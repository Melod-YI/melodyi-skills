原始需求：
我计划在一个python的agent项目里，增加一个读文件的工具。
这个agent项目属于通用agent（不是coding agent）。
agent实际运行的时候会跨平台。可在windows上直接通过python命令或者uv命令等启动。也可能在linux环境或者容器环境下运行。

  1. 工具概述

  def validate_params(path: str, offset: int, limit: int) -> 
  Optional[ReadError]:
      if not path or not path.strip():
          return ReadError(error="path is required", error_code="invalid_param")

      if offset < 1:
          return ReadError(error=f"offset must be >= 1, got {offset}",
  error_code="invalid_param")

      if limit < 1 or limit > MAX_LINES:
          return ReadError(error=f"limit must be between 1 and {MAX_LINES}",
  error_code="invalid_param")

  def normalize_path(path: str, workspace: str) -> str:
      # 1. trim
      path = path.strip()

      # 2. ~展开
      if path.startswith('~'):
          path = os.path.expanduser(path)

      # 3. 相对→绝对
      if not os.path.isabs(path):
          path = os.path.join(workspace, path)

      # 4. 跨平台规范化
      path = os.path.normpath(path)

      # 5. Null bytes检查 [P1]
      if '\0' in path:
          raise ValueError("Path contains null bytes")

      return path

  def is_blocked_device(path: str) -> bool:

      if path.startswith('/proc/') and path.endswith('/fd/0') or

      # 使用pathlib确保跨平台正确性
      try:

          workspace_obj = Path(workspace).resolve()

  检查项:

  ┌────────────────┬────────┬──────────────────────────────────┐
  │      检查      │ 优先级 │             错误处理             │

  相似文件建议算法:
  def suggest_similar_files(target_path: str, max_suggestions: int = 5) -> 
  List[str]:
      """
      当文件不存在时，提供相似文件建议
      
      算法：
      1. 获取同目录下的文件列表（排除隐藏文件）
      2. 计算目标文件名与每个候选文件的字符重叠度
      3. 返回重叠度 >= 50% 的前N个文件
      """
      dir_path = os.path.dirname(target_path)
      target_name = os.path.basename(target_path).lower()

      if not os.path.isdir(dir_path):
          return []

      candidates = []
      for entry in os.listdir(dir_path):
          if entry.startswith('.'):  # 排除隐藏文件
              continue
          entry_lower = entry.lower()

          # 双向包含检查（参考OpenCode）
          if target_name in entry_lower or entry_lower in target_name:
              candidates.append(os.path.join(dir_path, entry))
          else:
              # 字符重叠度检查（参考Hermes/OpenClaw）
              common = set(target_name) & set(entry_lower)
              overlap_ratio = len(common) / max(len(target_name), 1)
              if overlap_ratio >= 0.5:
                  candidates.append(os.path.join(dir_path, entry))

      return candidates[:max_suggestions]

  内容采样检测:
  def is_binary_by_content(file_path: str, sample_size: int = 1024) -> bool:
      """
      通过内容采样检测是否为二进制文件
      
      算法：
      1. 读取前N字节
      2. 发现 \0 字节 → 直接判定二进制
      3. 计算控制字符比例（排除 \n\r\t）
      4. 比例 > 30% → 二进制
      """
      try:
          with open(file_path, 'rb') as f:
              sample = f.read(sample_size)

          if not sample:
              return False  # 空文件不算二进制

          # 空字节直接判定
          if b'\x00' in sample:
              return True

          # 控制字符统计
          non_printable = sum(
              1 for b in sample
              if b < 32 and b not in (9, 10, 13)  # 排除 \t \n \r
          )

          return non_printable / len(sample) > 0.30

      except Exception:
          return False

  MAX_FILE_SIZE = 50 * 1024 * 1024  # 50MB
  LARGE_FILE_THRESHOLD = 512 * 1024  # 512KB

  def check_file_size(file_path: str) -> Tuple[int, Optional[str]]:
      """返回文件大小和可能的提示"""
      size = os.path.getsize(file_path)

      if size > MAX_FILE_SIZE:
          raise FileTooLargeError(
              f"File exceeds {MAX_FILE_SIZE // 1024 // 1024}MB limit. "
              f"Use offset and limit parameters to read specific sections."
          )

      hint = None
      if size > LARGE_FILE_THRESHOLD:
          hint = f"This file is large ({size // 1024}KB). Consider using 
  offset/limit for targeted reads."

      return size, hint

  ┌───────────────┬────────┬───────────────────────────────────────────────┐

  ├───────────────┼────────┼───────────────────────────────────────────────┤

  ├───────────────┼────────┼───────────────────────────────────────────────┤
  │ UTF-8 BOM剥离 │ P1     │ 如果开头是 0xEF 0xBB 0xBF，剥离BOM            │
  ├───────────────┼────────┼───────────────────────────────────────────────┤
  │ 编码失败回退  │ P1     │ 无法检测时，默认UTF-8（常见情况）             │
  └───────────────┴────────┴───────────────────────────────────────────────┘

  def detect_encoding(file_path: str) -> str:
      """
      检测文件编码
      
      使用 charset_normalizer 库（Python 3.9+ 内置推荐）
      """
      try:
          # 读取前10KB用于检测
          with open(file_path, 'rb') as f:
              sample = f.read(10 * 1024)

          # UTF-8 BOM检测
          if sample.startswith(b'\xef\xbb\xbf'):
              return 'utf-8-sig'  # BOM标记

          # charset_normalizer检测
          result = charset_normalizer.detect(sample)
          if result and result.get('encoding'):
              encoding = result['encoding']
              # 常见编码映射
              if encoding.lower() in ('ascii', 'utf-8', 'utf8'):
                  return 'utf-8'
              return encoding

      except Exception:
          pass

      # 默认UTF-8
      return 'utf-8'

  Layer 8: 内容读取 [P0-必须]

  读取策略:

  ┌──────────┬──────────┬─────────────────────────────────┐

  ├──────────┼──────────┼─────────────────────────────────┤
  │ 流式读取 │ 所有文件 │ 使用 open() + 逐行读取，避免OOM │
  └──────────┴──────────┴─────────────────────────────────┘

  实现要点:
  def read_lines(file_path: str, offset: int, limit: int, encoding: str) -> 
  Tuple[List[str], int]:
      """
      流式读取指定范围的行
      
      返回：(读取的行列表, 文件总行数)
      """
      lines = []
      total_lines = 0
      start_index = offset - 1  # 0-indexed
      end_index = start_index + limit

      with open(file_path, 'r', encoding=encoding, errors='replace') as f:
          for line_idx, line in enumerate(f):
              total_lines += 1

              # 跳过offset之前的行
              if line_idx < start_index:
                  continue

              # 超出limit，继续计数但不存储
              if line_idx >= end_index:
                  continue

              # 存储行内容（strip末尾换行符）
              lines.append(line.rstrip('\r\n'))

      return lines, total_lines

  Layer 9: 行内容处理 [P0-必须]

  处理项:

  ┌────────────┬────────┬────────────────────────────────────┐
  │    处理    │ 优先级 │                说明                │
  ├────────────┼────────┼────────────────────────────────────┤
  │ 单行截断   │ P0     │ 超过 MAX_LINE_LENGTH → 截断 + 标记 │
  ├────────────┼────────┼────────────────────────────────────┤
  │ CRLF转换   │ P0     │ \r\n → \n（已在Layer 8处理）       │
  ├────────────┼────────┼────────────────────────────────────┤
  │ 行号格式化 │ P0     │ `     N                            │
  └────────────┴────────┴────────────────────────────────────┘

  MAX_LINE_LENGTH = 2000
  LINE_TRUNCATION_SUFFIX = "... [line truncated at {MAX_LINE_LENGTH} chars]"

  def format_lines(lines: List[str], start_line: int) -> Tuple[str, List[int]]:
      """
      格式化行内容，返回格式化文本和截断的行号列表
      """
      formatted = []
      truncated_lines = []

      for idx, line in enumerate(lines):
          line_num = start_line + idx

          # 单行截断检查
          if len(line) > MAX_LINE_LENGTH:
              line = line[:MAX_LINE_LENGTH] + LINE_TRUNCATION_SUFFIX
              truncated_lines.append(line_num)

          # 行号格式化 "     1|content"
          formatted.append(f"{line_num:6d}|{line}")

      return '\n'.join(formatted), truncated_lines

  Layer 10: 输出大小检查 [P0-必须]

  检查项:

  ┌──────────────┬────────┬──────────────────────────────┬─────────────────┐
  │     检查     │ 优先级 │             阈值             │      处理       │
  ├──────────────┼────────┼──────────────────────────────┼─────────────────┤
  │ 输出字符限制 │ P0     │ MAX_OUTPUT_CHARS (默认100KB) │ 截断 + 提示续读 │
  └──────────────┴────────┴──────────────────────────────┴─────────────────┘

  MAX_OUTPUT_CHARS = 100_000  # 100KB

  def check_output_size(content: str, total_lines: int, current_end_line: int) 
  -> Tuple[str, bool, Optional[str]]:
      """
      检查输出大小，必要时截断
      
      返回：(可能截断的内容, 是否截断, 提示信息)
      """
      if len(content) <= MAX_OUTPUT_CHARS:
          return content, False, None

      # 截断到限制
      truncated_content = content[:MAX_OUTPUT_CHARS]

      # 计算实际输出的行数（估算）
      lines_in_output = truncated_content.count('\n')
      actual_end_line = current_end_line - (total_lines - lines_in_output)

      hint = (
          f"(Output capped at {MAX_OUTPUT_CHARS // 1024}KB. "
          f"Showing lines up to ~{actual_end_line}. "
          f"Use offset={actual_end_line + 1} to continue.)"
      )

      return truncated_content, True, hint

      # Slack

      # Google
      r'AIza[a-zA-Z0-9_-]{35}',         # Google API key

      # AWS
      r'AKIA[A-Z0-9]{16}',              # AWS access key
      r'ASIA[A-Z0-9]{16}',              # AWS temporary key

      # Stripe
      r'sk_live_[a-zA-Z0-9]{24,}',      # Stripe live key
      r'sk_test_[a-zA-Z0-9]{24,}',      # Stripe test key

      # Generic patterns
      r'(api[_-]?key|apikey)["\s:=]+["\']?[a-zA-Z0-9]{20,}',
      r'(secret|token|password|passwd|pwd)["\s:=]+["\']?[a-zA-Z0-9]{10,}',
  ]

  # 私钥块
  PRIVATE_KEY_PATTERN = r'-----BEGIN (?:RSA |DSA |EC |OPENSSH )?PRIVATE 
  KEY-----'

  # 数据库连接串
  DB_CONN_PATTERN = r'(mysql|postgres|mongodb|redis)://[^\s]+:[^\s]+@[^\s]+'

  # Authorization header
  AUTH_HEADER_PATTERN = r'Authorization:\s*(Bearer|Basic)\s+[a-zA-Z0-9._-]+'

  脱敏函数:
  def redact_sensitive_text(text: str) -> Tuple[str, bool]:
      """
      脱敏敏感信息
      
      返回：(脱敏后的文本, 是否有脱敏)
      """
      redacted = False
      result = text

      # 前缀模式脱敏
      for pattern in SENSITIVE_PREFIXES:
          matches = re.findall(pattern, result, re.IGNORECASE)
          for match in matches:
              masked = mask_token(match)
              result = result.replace(match, masked)
              redacted = True

      # 私钥块脱敏
      if re.search(PRIVATE_KEY_PATTERN, result):
          result = re.sub(
              PRIVATE_KEY_PATTERN + r'.*?' + r'-----END.*?PRIVATE KEY-----',
              '[REDACTED PRIVATE KEY]',
              result,
              flags=re.DOTALL
          )
          redacted = True

      # 数据库连接串脱敏
      db_matches = re.findall(DB_CONN_PATTERN, result, re.IGNORECASE)
      for match in db_matches:
          result = result.replace(match, '[REDACTED DATABASE CONNECTION]')
          redacted = True

      return result, redacted

  def mask_token(token: str) -> str:
      """
      Token脱敏规则：
      - 短token (<18字符): ***
      - 长token: 保留前6 + 后4 → sk-abc...xyz
      """
      if len(token) < 18:
          return '***'
      return f"{token[:6]}...{token[-4:]}"

  Layer 12: 截断提示生成 [P1-重要]

  ┌──────────────────┬────────┬────────────────────┐
  │     提示类型     │ 优先级 │      生成条件      │
  ├──────────────────┼────────┼────────────────────┤
  │ 续读提示         │ P1     │ 行数截断或大小截断 │
  ├──────────────────┼────────┼────────────────────┤
  │ 单行截断工具建议 │ P1     │ 有单行被截断       │

  │ 大文件建议       │ P2     │ 文件 > 512KB       │
  └──────────────────┴────────┴────────────────────┘

  def generate_hints(
      truncated: bool,
      truncated_lines: List[int],
      total_lines: int,
      end_line: int,
      file_size: int,
  ) -> Optional[str]:
      """
      生成截断提示
      """
      hints = []

      # 续读提示
      if truncated and end_line < total_lines:
          hints.append(
              f"(Showing lines {end_line - len(truncated_lines) + 1}-{end_line} 
  of {total_lines}. "
              f"Use offset={end_line + 1} to continue.)"
          )

      # 单行截断工具建议
      if truncated_lines:
          hints.append(
              f"⚠ Lines {truncated_lines} were truncated. For compressed 
  files:\n"
              f"  • JSON: Use bash tool with 'jq .' to format\n"
              f"  • Search: Use grep tool for specific patterns"
          )

      # 大文件建议 [P2]
      if file_size > LARGE_FILE_THRESHOLD:
          hints.append(
              f"💡 File size: {file_size // 1024}KB. Use offset/limit for 
  targeted reads."
          )

      return '\n'.join(hints) if hints else None

          return path

  8.2 主函数骨架

  # tools/read_file/tool.py

  async def read_file(
      path: str,
      offset: int = 1,
      limit: int = 500,
  ) -> Dict[str, Any]:
      """
      读取文本文件
      
      Args:
          path: 文件路径（绝对/相对/~格式）
          offset: 起始行号（1-indexed）
          limit: 读取行数
      
      Returns:
          成功: {"content": str, "total_lines": int, ...}
          失败: {"error": str, "error_code": str, "similar_files": list}
      """
      config = get_read_file_config()  # 获取配置

      try:
          # Layer 1: 参数验证
          param_error = validate_params(path, offset, limit)
          if param_error:
              return error_to_dict(param_error)

          # Layer 2-3: 路径预处理 + 安全守卫
          normalized_path = normalize_path(path, config.workspace)
          security_error = check_security(normalized_path, config)
          if security_error:
              return error_to_dict(security_error)

          # Layer 4-5: 文件存在性 + 二进制检测
          exists_error = check_file_exists(normalized_path)
          if exists_error:
              return error_to_dict(exists_error)

          if is_binary(normalized_path):
              return error_to_dict(ReadError(
                  error=f"Binary file detected: {path}",
                  error_code="binary_file"
              ))

          # Layer 6-7: 文件大小 + 编码检测
          file_size, size_hint = check_file_size(normalized_path, config)
          encoding = detect_encoding(normalized_path)

          # Layer 8-9: 内容读取 + 格式化
          lines, total_lines = read_lines(normalized_path, offset, limit,
  encoding)
          formatted_content, truncated_line_nums = format_lines(lines, offset)

          # Layer 10: 输出大小检查
          content, output_truncated, output_hint = check_output_size(
              formatted_content, total_lines, offset + len(lines)
          )

          # Layer 11: 敏感信息脱敏
          if config.enable_redaction:
              content, was_redacted = redact_sensitive_text(content)
              redaction_warning = "⚠ Content was redacted for sensitive data" if
   was_redacted else None

          # Layer 12: 提示生成
          hint = generate_hints(
              truncated=output_truncated or len(lines) < limit,
              truncated_lines=truncated_line_nums,
              total_lines=total_lines,
              end_line=offset + len(lines),
              file_size=file_size,
          )

          # Layer 13: 返回结果
          return {
              "content": content,
              "total_lines": total_lines,
              "file_size": file_size,
              "encoding": encoding,
              "truncated": output_truncated or len(lines) < limit,
              "hint": hint,
              "warning": redaction_warning,
          }

      except Exception as e:
          return error_to_dict(ReadError(
              error=str(e),
              error_code="unexpected_error"
          ))
