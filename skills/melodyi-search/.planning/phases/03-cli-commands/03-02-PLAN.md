# 示例: 20260509-143052-a1b2
def generate_session_id() -> str:
    timestamp = datetime.now().strftime("%Y%m%d-%H%M%S")
    random_suffix = ''.join(random.choices('0123456789abcdef', k=4))
    return f"{timestamp}-{random_suffix}"
```

    """Comparison 模式 E2E 验证测试"""

    import os
    import sqlite3
    import tempfile
    from pathlib import Path
    from unittest.mock import MagicMock

    from melodyi_search.domain.services.execution_strategy import ExecutionStrategy
    from melodyi_search.domain.services.comparison_recorder import ComparisonRecorder
    from melodyi_search.infrastructure.database.database_manager import DatabaseManager
    from melodyi_search.infrastructure.config.config_schema import DatabaseConfig
    from melodyi_search.providers.base_provider import BaseProvider, ProviderSearchRequest, ProviderSearchResult
    from melodyi_search.domain.models.search_result import SearchResultItem

    class TestComparisonE2E:
        """Comparison 模式端到端测试"""

        def test_comparison_full_flow(self):
            """测试完整 comparison 流程"""
            # 创建临时数据库
            with tempfile.TemporaryDirectory() as tmpdir:
                db_path = Path(tmpdir) / "test.db"
                config = DatabaseConfig(database_path=str(db_path))
                db_manager = DatabaseManager(config)
                db_manager.init_database()

                # 创建 recorder
                recorder = ComparisonRecorder(db_manager)

                # 创建 mock providers
                mock_providers = [
                    self._create_mock_provider("provider1", [
                        SearchResultItem(title="Result 1", url="https://example.com/1")
                    ]),
                    self._create_mock_provider("provider2", [
                        SearchResultItem(title="Result 2", url="https://example.com/2")
                    ])
                ]

                # 执行 comparison
                strategy = ExecutionStrategy()
                request = ProviderSearchRequest(query="test query")
                result = strategy.execute_comparison(mock_providers, request, recorder)

                # 验证 session_id
                assert result.session_id is not None
                assert self._validate_session_id_format(result.session_id)

                # 验证数据库记录
                conn = db_manager.get_connection()
                try:
                    # 验证 comparison_sessions
                    sessions = conn.execute(
                        "SELECT * FROM comparison_sessions WHERE session_id = ?",
                        (result.session_id,)
                    ).fetchall()
                    assert len(sessions) == 1
                    assert sessions[0][1] == "test query"  # query

                    # 验证 provider_results
                    providers = conn.execute(
                        "SELECT * FROM provider_results WHERE session_id = ?",
                        (result.session_id,)
                    ).fetchall()
                    assert len(providers) == 2  # 两个供应商
                    assert providers[0][1] == "provider1"
                    assert providers[1][1] == "provider2"

                    # 验证 search_results
                    results = conn.execute(
                        "SELECT * FROM search_results WHERE session_id = ?",
                        (result.session_id,)
                    ).fetchall()
                    assert len(results) == 2  # 两个结果
                finally:
                    conn.close()

        def _create_mock_provider(self, name: str, results: list) -> BaseProvider:
            """创建 mock provider"""
            mock = MagicMock(spec=BaseProvider)
            mock.name = name
            mock.search.return_value = ProviderSearchResult(
                provider=name,
                results=results,
                response_time_ms=100,
                raw_response={}
            )
            return mock

        def _validate_session_id_format(self, session_id: str) -> bool:
            """验证 session_id 格式: YYYYMMDD-HHMMSS-XXXX"""
            import re
            pattern = r'^\\d{8}-\\d{6}-[0-9a-f]{4}$'
            return re.match(pattern, session_id) is not None
    ```
